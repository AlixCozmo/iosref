--- layout: default title: Swift 4 cheatsheet ---

Variables

{% highlight swift %} let myInt = 5 let myString = "5.5" {% endhighlight %}

Type annotations

{% highlight swift %} let myInt: Int = 5 let myString: String = "5.5" {% endhighlight %}

Mutable variables

{% highlight swift %} let myInt1 = 5 // Can't change later var myInt2 = 8 // Can change later {% endhighlight %}

If statement

{% highlight swift %} let myBool = true if myBool { print("myBool is true") } else { print("myBool is false") } // Output: "myBool is true" {% endhighlight %}

Optionals

{% highlight swift %} let myInt: Int? = 5 if let unwrappedInt = myInt { print("myInt is \(unwrappedInt)") } else { print("myInt is nil") } // Output: "myInt is 5" {% endhighlight %}

Enum

{% highlight swift %} enum CompassPoint { case north case south case east case west } {% endhighlight %}

Enum variable

{% highlight swift %} var direction: CompassPoint = .north {% endhighlight %}

Switch statement

{% highlight swift %} switch direction { case .north: print("Going north!") case .south: print("Going south...") default: print("Going east or west.") } // Output: "Going north!" {% endhighlight %}

Function with param and return

{% highlight swift %} func square(x: Int) -> Int { let squaredValue = x * x return squaredValue } {% endhighlight %}

Calling a functions

{% highlight swift %} let a = 6 let squareOfA = square(x: a) print("Square of a is: \(squareOfA)") // Output: "Square of a is: 36" {% endhighlight %}

Creating a struct

{% highlight swift %} struct MyStruct { var myInt = 5 let myOptString: String? init(str: String) { self.myOptString = str } mutating func squareMyInt() { myInt = myInt * myInt } } {% endhighlight %}

Array

{% highlight swift %} var myArr = [1, 3] myArr.append(5) {% endhighlight %}

Loop over array

{% highlight swift %} for element in myArr { print("Element: \(element)") } /* Output: Element: 1 Element: 3 Element: 5 */ {% endhighlight %}
